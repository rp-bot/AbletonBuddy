\section{Introduction}

\begin{frame}\frametitle{\secname}\framesubtitle{Overview}
\begin{columns}[T,totalwidth=\textwidth]
\column{0.55\textwidth}
\begin{itemize}
    \item<1-> \textbf{AbletonBuddy} is an AI-powered assistant that enables natural language control of Ableton Live.
    \item<2-> Uses large language models (LLMs) to understand user commands and translate them into OSC (Open Sound Control) commands.
    \item<3-> Provides an intuitive interface for controlling complex DAW operations through conversational interaction.
    \item<4-> Supports both real-time control and simulation mode for testing and development.
\end{itemize}

\begin{frame}\frametitle{\secname}\framesubtitle{Key Features}
\begin{itemize}
    \item<1-> \textbf{Natural Language Interaction}: Control Ableton Live with simple, conversational commands
    \item<2-> \textbf{Comprehensive Control}: Manage tracks, scenes, clips, devices, and playback
    \item<3-> \textbf{Conversation Persistence}: All interactions saved to SQLite database for continuity
    \item<4-> \textbf{Real-time Streaming}: Watch the assistant work in real-time via streaming API
    \item<5-> \textbf{Modern Web Interface}: React-based chat UI with real-time updates
    \item<6-> \textbf{Extensible Architecture}: Easy to add new tools and capabilities
\end{itemize}
\end{frame}
