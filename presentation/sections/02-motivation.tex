\begin{frame}\frametitle{\secname}\framesubtitle{Why Natural Language?}
\begin{itemize}
    \item<1-> \textbf{Intuitive Interaction}: Users can express intent naturally without learning specific commands
    \item<2-> \textbf{Contextual Understanding}: LLMs can infer user intent from ambiguous requests
    \item<3-> \textbf{Learning Aid}: Helps users discover Ableton Live capabilities through conversation
    \item<4-> \textbf{Future-Proof}: Natural language interface adapts as LLM capabilities improve
    \item<5-> \textbf{Multimodal Potential}: Foundation for future audio-visual understanding and control
\end{itemize}
\end{frame}
